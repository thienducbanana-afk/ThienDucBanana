local Players = game:GetService("Players")
local DataStoreService = game:GetService("DataStoreService")

local banStore = DataStoreService:GetDataStore("OneDayBanStore")

local BAN_DURATION = 86400 -- 1 ngày (giây)

Players.PlayerAdded:Connect(function(player)

	local userId = player.UserId
	local currentTime = os.time()

	local banData
	local success, err = pcall(function()
		banData = banStore:GetAsync(userId)
	end)

	if success and banData then
		if currentTime < banData then
			local remaining = banData - currentTime
			local hours = math.floor(remaining / 3600)
			player:Kick(" Bn ã b banned. Thi gian còn li: "..hours.." gi.")
			return
		end
	end

	--  T NG BAN 1 NGÀY KHI VÀO
	local expireTime = currentTime + BAN_DURATION

	pcall(function()
		banStore:SetAsync(userId, expireTime)
	end)

	player:Kick(" Bn ã b banned 1 ngày!")
end)
